my $cnt = 1;
open (IN, "find '/Volumes/iPhoto\ Library\ -\ 4\:5\:05' -iname '*.jpg' |"); 
while (<IN>) {
   chomp;
   next if (/originals/i);
   next if (/thumbs/i);
   my $full_path = $_;
   my $file = $_;
   $file =~ s/.*\///;
   $cmd = "cp '$full_path' lynn_photos/$cnt-$file";
   print "$cmd\n";
   `$cmd`;
   $cnt++;
}
close IN;

Quick and dirty IPv4 sorter?

my @ips = qw( 20.0.100.0 20.0.50.0 100.0.0.0 77.0.0.0 );

print join ", ", sort @ips;
print "\n";
print join ", ", sort by_ip @ips;
print "\n";

sub by_ip {
   my ($j, $k) = ($a, $b);
   for ($j, $k) {
      $_ = sprintf("%03d.%03d.%03d.%03d", split /\./);
   }
   $j cmp $k;
}

First example I found...

    my ($common_rate,$cnt);
    foreach my $rate (keys %common_rates) {
      if ($common_rates{$rate} >= $cnt) {
        next if ($rate < $common_rate);
        $common_rate = $rate;
        $cnt = $common_rates{$rate};
      }
    }

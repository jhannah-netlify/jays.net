Been using this 'pmpath' script for years...

  #!/usr/local/bin/perl
  $module = shift;
  ($mod = $module) =~ s#::#/#g;
  die ("Need a module name\n") unless $mod;
  $mod .= '.pm';
  require $mod;
  print $INC{$mod} . " (" . ${$module . "::VERSION"} . ")\n";

My version was a lot longer (Reverse Perl Golf!!):

   my @files = @{$self->{files}};
   my $file;
   foreach $file (@files) {
      open (IN, $file);
      my $locus;
      while (<IN>) {
         if (/^LOCUS/) {
            # A locus has begun.
            $locus = $_;
         } elsif (/^\/\//) {
            # A locus ends.
            $locus .= $_;
            if ($locus =~ /$args{grep}/s) {
               print OUT $locus;
            }
         } else {
            # A row inside a locus.
            $locus .= $_;
         }
      }
   }

to prove it worked.  :)

$ cat j.pl

my $apply_this_card_to_all = "new";

my $before = <<EOT;
...xxx...
...xxx...
...xxx...
EOT

my $after = $before;
$after =~ s/(.{3}).{3}(.*?\n)/$1$apply_this_card_to_all$2/gs;
print "$before$after";


$ perl j.pl

...new...
...new...

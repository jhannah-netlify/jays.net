Task: Find any lines in a pipe-delimited file where the 3rd column contains a non-alphanumeric character. 

j


#!/usr/bin/perl

open (IN, "j.unl");
while (<IN>) {
   chomp;
   @l = split /\|/;
   print if ($l[2] =~ /[^a-z0-9]/i);
}

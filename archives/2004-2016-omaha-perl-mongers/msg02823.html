> Oh. Huh. Here's a simple outlier removal system.
> 
> https://metacpan.org/module/Statistics::Basic#COMPUTED-VECTORS
> 
> :)
> 
> j
> 
> 
> 
> 
> $ cat j.pl
> use Statistics::Basic qw( :all );
> 
> my @a = ( (1,2,3) x 7, 15 );
> 
> my $v1 = vector(@a);
> my $v3 = computed($v1);
> $v3->set_filter(sub {
>   my $m = mean($v1);
>   my $s = stddev($v1);
>   grep { abs($_-$m) <= $s } @_;
> });
> 
> print "$v1\n";
> print "$v3\n";
> 
>

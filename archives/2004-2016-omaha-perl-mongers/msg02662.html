         my $skill = $c->req->param("skill_id_$i");
         my $score = $c->req->param("score_id_$i");
         last unless ($skill);
         last if ($skill =~ /^-- /);
         push @where_skills, {
            skill_id => $skill,
            score => { '>=', $score },
         };
      }
      my $where = {
         active => 1,
         -or => [ @where_skills ],
      };
      $c->log->debug(Dumper($where));
      $search = $c->model('DB::Rating')->search($where);

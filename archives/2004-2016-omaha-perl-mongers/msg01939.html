If you put parentheses in the right place you get what you want.

cat t.pl
print (((not defined $j) || 1) ? 'yes ' : 'no ');
print (((not defined $j) or 1) ? 'yes ' : 'no ');
print ((! defined $j || 1) ? 'yes ' : 'no ');
print ((not defined $j or 1) ? 'yes ' : 'no ')
